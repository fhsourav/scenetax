scenetax newproject <projectname> // creates a new project

scenetax project --archive // archives the entire project as .tar.gz
scenetax project --commit // adds the changes and commits with automated commit message
scenetax project compile --epub // compiles as epub using pandoc
scenetax project compile --pdf // compiles as pdf using pandnoc

scenetax project character --create <name> // creates a character with the specified name
scenetax project create --location <name> // creates a location with the specified name
scenetax project create --group <name> // creates a group with the specified name

scenetax project write --scene [--newchapter] // looks inside the drafts, creates a new scene in the last chapter sorting in ascending order. if --newchapter is used, creates a new chapter folder and creates the scene file.
